#include<bits/stdc++.h>
using namespace std;


struct Node{
    int data;
    struct Node *next;
    struct Node *prev;
};

struct Node *head = NULL;
struct Node *tail = NULL;


void insertAtBegin(int data){

    struct Node *new_node = new Node();
    new_node->data = data;
    new_node->next = NULL;
    new_node->prev = NULL;


    if(head == NULL && tail == NULL){ // new
        head = new_node;
        tail = new_node; // new
        return;
    }

    new_node->next = head;
    head->prev = new_node;
    head = new_node;
}


void insertAtEnd(int data){

    struct Node *new_node = new Node();
    new_node->data = data;
    new_node->next = NULL;
    new_node->prev = NULL;


    if(head == NULL && tail == NULL){ // new
        head = new_node;
        tail = new_node; // new
        return;
    }

    struct Node *ptr;
    ptr = tail; // new

    ptr->next = new_node;
    new_node->prev = ptr;
    tail = new_node;
}


void deleteFromBegin(){

    if(head != NULL){
        struct Node *ptr;
        ptr = head;

        // If only one node exists
        if(head->next == NULL){
            head = NULL;
            tail = NULL;
        }
        else{
            head = head->next;
            head->prev = NULL;
        }

        free(ptr);
    }
    else if(tail != NULL){
        struct Node *ptr;
        ptr = tail;

        // If only one node exists
        if(tail->prev == NULL){
            head = NULL;
            tail = NULL;
        }
        else{
            tail = tail->prev;
            tail->next = NULL;
        }

        free(ptr);
    }
}


void deleteFromEnd(){
    if(head == NULL){
            cout << "The List is empty";
            return;
    }
    struct Node *ptr;
    ptr = head;
    while(ptr->next != NULL){
            ptr = ptr->next;
    }
    ptr->prev->next = NULL;
    free(ptr);
}

void sortedInsertion(int data){
    struct Node *new_node = new Node();
    new_node->data = data;
    new_node->next = NULL;
    new_node->prev = NULL;


    if(head == NULL || head->data > new_node->data){
            new_node->next = head;
            head->prev = new_node;
            head = new_node;
            return;
    }
    struct Node *ptr;
    ptr = head;
    while(ptr!=NULL && ptr->data < new_node->data){
        ptr = ptr->next;
    }
    new_node->next = ptr->next;
    ptr->next->prev = new_node;
    new_node->prev = ptr;
    ptr->next = new_node;
}


void insertAtGivenPosition(int data, int pos){

    struct Node *new_node = new Node();
    new_node->data = data;
    new_node->next = NULL;
    new_node->prev = NULL;

    if(pos == 1){
            new_node->next = head;
            head->prev = new_node;
            head = new_node;
            return;
    }

    struct Node *ptr;
    ptr = NULL;
    for(int i=1;i<pos-1 && ptr!=NULL; i++){
        ptr = ptr->next;
    }

    if(ptr == NULL){
        cout << "Invalid Position";
        return;
    }

    new_node->next = ptr->next;
    ptr->next->prev = new_node;
    new_node->prev = ptr;
    ptr->next = new_node;

}

stack<char> S;

int priority(char s){
    if(s == '^') return 100;
    else if(s == '*' || s == '/') return 50;
    else if(s == '+' || s == '-') return 20;
    else return 0;
}
